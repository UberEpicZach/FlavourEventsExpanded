add_good_clerical_education = {
	add_ruler_modifier = {
		name = good_clerical_education
		duration = -1
	}
	if = {
		limit = {
			NOT = {
				adm = 6
			}
		}
		change_adm = 1
	}
	else = {
		add_ruler_modifier = {
			name = adm_education_bonus_1
			duration = -1
		}
	}
}
add_best_clerical_education = {
	add_ruler_modifier = {
		name = best_clerical_education

	}
	if = {
		limit = {
			NOT = {

		}
