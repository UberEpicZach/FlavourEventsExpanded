fee_estate_privilege_coastline_expansion = {
	icon = privilege_dev_cost_desert
	max_absolutism = -10
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 2 } }
	influence = 0.05
	loyalty = 0.05
	is_valid = {
		OR = {
			has_unlocked_estate_privilege = { estate_privilege = fee_estate_privilege_coastline_expansion }
		}
	}
	can_select = {
		any_owned_province = {
			OR = {
				has_terrain = coastline
			}
		}
	}
	on_granted = {
		custom_tooltip = fee_estate_privilege_coastline_expansion_gt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_terrain = coastline
				}
				add_province_modifier = {
					name = estate_privilege_coastline_expansion
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = fee_estate_privilege_coastline_expansion_rt
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = estate_privilege_coastline_expansion
					}
				}
				remove_province_modifier = estate_privilege_coastline_expansion
			}
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 2
			num_of_owned_provinces_with = {
				OR = {
					has_terrain = coastline
				}
				value = 8
			}
		}
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}
}
