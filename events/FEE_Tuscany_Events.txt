#Freedom of Religion

	title = FEE_Tuscany_Events.2.T
	desc = FEE_Tuscany_Events.2.D
	picture = RELIGION_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = LAN
			tag = TUS

			was_tag = TUS
		}
		NOT = { has_idea_group = religious_ideas }
		is_year = 1550
		NOT = { is_year = 1585 }
		religion = catholic
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.75
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 0.75
			any_owned_province = {
				OR = {
					religion = protestant
					religion = reformed
				}
			}
		}
		modifier = {
			factor = 0.8
			ADM = 4
		}
		modifier = {
			factor = 1.3
			NOT = {
				ADM = 3
			}
		}
	}

	option = {
		name = FEE_Tuscany_Events.2.OPT1

		add_adm_power = -25
		add_country_modifier = {
			name = fee_tus_laws_freedom_religion
			duration = 7300
		}

		capital_scope = {
			add_base_tax = 1
			add_base_production = 1
		}
	}

	option = {
		name = FEE_Tuscany_Events.2.OPT2

		add_years_of_income = 0.5
		add_country_modifier = {
			name = fee_tus_one_duchy_one_religion
			duration = 7300
		}
	}
}

#Emperor Condemn's the Duchy's Elevation
country_event = {
	id = FEE_Tuscany_Events.3
	title = FEE_Tuscany_Events.3.T
	desc = FEE_Tuscany_Events.3.D
	picture = LIBERUM_VETO_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = LAN
			tag = TUS
		}

					who = emperor
					value = 75
				}

				num_of_cities = 3
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = fee_grand_duchy_reaction
				duration = -1
			}
		}
	}

	option = {
		name = FEE_Tuscany_Events.3.OPT1

		add_dip_power = -25
		add_adm_power = 25
		change_religious_influence_equivalent_fee = { add = 15 }
		add_prestige = 10
	}
}

	id = FEE_Tuscany_Events.10
	title = FEE_Tuscany_Events.10.T
	desc = FEE_Tuscany_Events.10.D
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = LAN
			tag = TUS
			was_tag = LAN
			was_tag = TUS
		}
		is_year = 1760
		NOT = { is_year = 1790 }
		is_at_war = no
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			has_idea_group = innovativeness_ideas
		}
		modifier = {
			factor = 1.1
			NOT = {
				ADM = 3
			}
		}
	}

	option = {
		name = FEE_Tuscany_Events.10.OPT1
		add_stability_or_adm_power = yes

		add_country_modifier = {
			name = fee_tus_liberal_reforms
			duration = -1
		}
		if = {
			limit = { has_country_flag = had_liberalism }
			country_event = {
				id = aspiration_for_liberty.4 days = 1
			}
		}
		if = {
			limit = {
				NOT = { has_country_flag = had_liberalism }
			}
			set_country_flag = had_liberalism
		}
	}

	option = {
		name = FEE_Tuscany_Events.10.OPT2

		add_stability_or_adm_power = yes
		add_adm_power = 100
		add_legitimacy_equivalent = { amount = 10 republican_tradition = 5 }
	}
}
