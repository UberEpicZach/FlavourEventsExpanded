	id = FEE_Navarra_Events.11
	title = FEE_Navarra_Events.11.T
	desc = FEE_Navarra_Events.11.D
	picture = FORT_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = NAV
			AND = {
				tag = SPA
				NOT = { exists = NAV }
			}
			AND = {
				tag = ARA
				NOT = { exists = NAV }
			}
		}
		owns = 210
		is_year = 1570
		NOT = { is_year = 1630 }
		exists = FRA
		NOT = { alliance_with = FRA }
		NOT = {
			has_opinion = {
				who = FRA
				value = 50
			}
		}
	}

	mean_time_to_happen = {
		months = 420
		modifier = {
			factor = 0.7
			truce_with = FRA
		}
		modifier = {
			factor = 0.8
			is_rival = FRA
		}
		modifier = {
			factor = 0.8
			FRA = {
				is_rival = ROOT
			}
		}
	}

	option = { #Build a great fort!
		name = FEE_Navarra_Events.11.OPT1
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				num_of_loans = 5
			}
		}

		add_years_of_income = -0.25
		210 = {
			add_province_modifier = {
				name = fee_nav_ciudadela_pamplona
				duration = -1
			}
		}
		if = {
			limit = {
				NOT = {
					210 = {
						fort_level = 2
					}
				}
			}

			}
		}
	}

	option = { #Let's develop a different strategy
		name = FEE_Navarra_Events.11.OPT2

			factor = 10
			modifier = {
				factor = 100
				num_of_loans = 5
			}
		}

		add_mil_power = 20
	}
}
