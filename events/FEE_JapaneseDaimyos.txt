# The Three Arrows
country_event = {
	id = fee_jpn_min.1
	title = "jpn_min_1_title"
	desc = "jpn_min_1_desc"
	fire_only_once = yes
	picture = GOOD_WITH_MONARCH_eventPicture

	trigger = {
		tag = MRI
		num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
			value = 5
		}
		is_year = 1550
		dynasty = "Mori"
		adm = 5
		has_heir = yes
	}

	mean_time_to_happen = { months = 50 }
	
	option = {
		name = "jpn_min_1_a"
		
		add_stability_or_adm_power = yes
		if = {
			limit = {
				NOT = {
					heir_adm = 6
				}
			}
			change_heir_adm = 1
		}
		else = {
			add_adm_power = 50
		}
	}
}
