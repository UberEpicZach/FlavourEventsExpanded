#Alliance oiffer to Lithuania
country_event = {
	id = fee_NOV.5
	title = fee_NOV.5.T
	desc = fee_NOV.5.D
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = fee_NOV.5.OPT1
		ai_chance = {
			factor = 100
		}
		create_alliance = FROM
		add_opinion = {
			who = FROM
			modifier = fee_NOV_gift
		}
		FROM = {
			add_opinion = {
				who = LIT
				modifier = fee_NOV_gift
			}
		}
	}
	
	option = {
		name = fee_NOV.5.OPT2
		ai_chance = {
			factor = 0
		}
		NOV = {
			add_casus_belli = {
				target = LIT
				type = cb_insult
				months = 300
			}
		}
	}
}
