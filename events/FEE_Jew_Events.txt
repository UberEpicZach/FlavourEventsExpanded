namespace = FEE_Jew_Events

#Medic at Court
country_event = {
	id = FEE_Jew_Events.1000
	title = FEE_Jew_Events.1000.T
	desc = FEE_Jew_Events.1000.D
	picture = ADVISOR_eventPicture
	
	trigger = {
		check_variable = {
			which = FEE_JEW_APPROVE
			value = 50
		}
		OR = {
			NOT = { has_country_flag = fee_jews_medic }
			had_country_flag = {
				flag = fee_jews_medic
				days = 5475
			}
		}
		NOT = {
			has_ruler_modifier = fee_jews_medic
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = fee_jews_medic
		}
	}

	mean_time_to_happen = {
		years = 25
		modifier = {
			factor = 1.25
			religious_ideas = 3
		}
	}

	option = {
		name = FEE_Jew_Events.1000.OPT1

		add_ruler_modifier = {
			name = fee_jews_medic
			duration = -1
		}
	}
}
