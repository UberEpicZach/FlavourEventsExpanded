# Kronstadt Fortress.
country_event = {
	id = fee_RUS.1
	title = "fee_RUS_n_1"
	desc = "fee_RUS_d_1"
	picture = CITY_VIEW_eventPicture
	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 300
	}
	
	trigger = {
		tag = RUS
		has_country_flag = relocated_capital_st_petersburg
		owns = 295
		owns = 33
		33 = {
			is_capital = yes
		}
	}
	
	option = {
		name = "fee_RUS_1_1"
		ai_chance = { factor = 70 }
		
		# Kronstadt Fortress modifier
		33 = {
			add_province_modifier = {
				name = "kronstadt_fortress"
				duration = -1
			}
		}
		
		# Grand Shipyard check.
		33 = {
			if = {
				limit = {
					NOT = { has_shipyard_building_trigger = yes }
				}
				
				add_building = grand_shipyard
			}
			else_if = {
				limit = {
					NOT = {
						has_building = grand_shipyard
					}
					has_building = shipyard
				}
				
				remove_building = shipyard
				add_building = grand_shipyard
			}
		}
		
		# Naval Battery check.
		33 = {
			if = {
				limit = {
					NOT = { has_building = coastal_defence }
					NOT = { has_building = naval_battery }
				}
				
				add_building = naval_battery
			}
			else_if = {
				limit = {
					NOT = {
						has_building = naval_battery
					}
					
					has_building = coastal_defence
				}
				
				remove_building = coastal_defence
				add_building = naval_battery
			}
		}
		
		# Fort check.
		33 = {
			if = {
				limit = {
					NOT = {
						fort_level = 1
					}
				}
				
				add_building = fort_15th
			}
			else_if = {
				limit = {
					NOT = {
						has_building = fort_16th
					}
					
					has_building = fort_15th
				}
				
				remove_building = fort_15th
				add_building = fort_16th
			}
			else_if = {
				limit = {
					NOT = {
						has_building = fort_17th
					}
					
					has_building = fort_16th
				}
				
				remove_building = fort_16th
				add_building = fort_17th
			}
			else_if = {
				limit = {
					NOT = {
						has_building = fort_18th
					}
					
					has_building = fort_17th
				}
				
				remove_building = fort_17th
				add_building = fort_18th
			}
		}
		
		add_years_of_income = -0.75
	}
	
	option = {
		name = "fee_RUS_2_1"
		ai_chance = { factor = 30 }
		
		# Renovated Kremlin modifier
		295 = {
			add_province_modifier = {
				name = "renovated_kremlin"
				duration = -1
			}
		}
		
		add_years_of_income = -0.5
	}
}
