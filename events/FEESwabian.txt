# Events for the formation, support and conflicts of the Swabian League


namespace = hee_swabian_events

# Establish a Defensive Swabian League

country_event = {
	id = hee_swabian_events.1
	title = hee_swabian_events.1.t
	desc = hee_swabian_events.1.d
	picture = LIBERUM_VETO_eventPicture

	trigger = {
		NOT = { has_global_flag = hee_swabian_league_established }
		OR = {
			capital_scope = { area = lower_swabia_area }
			capital_scope = { area = upper_swabia_area }
		}
		is_year = 1460
		NOT = { is_year = 1520 }
		NOT = { has_global_flag = hee_swabian_league_disbanded }
		NOT = { is_at_war = yes }
		NOT = { tag = BAV }
		is_part_of_hre = yes
		is_subject = no
		NOT = {
			num_of_owned_provinces_with = {
				value = 4
				NOT = { 
					OR = {
						area = lower_swabia_area
						area = upper_swabia_area
					} 
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			has_truce = yes
		}
	}

	option = {
		name = hee_swabian_events.1.a		# Establish a league for all of Swabia!
		add_country_modifier = {
			name = hee_joined_swabian_league
			duration = -1
		}
		set_global_flag = hee_swabian_league_established
		add_prestige = 10
	}
	
	option = {
		name = hee_swabian_events.1.b		# We fight on our own.
		add_mil_power = 10
		add_prestige = 5
	}
}

# Asked if country wants to join

country_event = {
	id = hee_swabian_events.2
	title = hee_swabian_events.2.t
	desc = hee_swabian_events.2.d
	picture = DIPLOMACY_eventPicture


	trigger = {
		OR = {
			OR = {
				capital_scope = { area = lower_swabia_area }
				capital_scope = { area = upper_swabia_area }
			}
			AND = {
				any_neighbor_country = {
					OR = {
						capital_scope = { area = lower_swabia_area }
						capital_scope = { area = upper_swabia_area }
					}
				}
				NOT = { tag = BAV }
				NOT = { tag = SWI }
				is_emperor = no
			}
		}
		has_global_flag = hee_swabian_league_established
		NOT = {
			has_country_modifier = hee_joined_swabian_league
		}
		NOT = {
			has_country_modifier = hee_rejected_swabian_league
		}
		NOT = {
			has_global_flag = hee_swabian_league_disbanded
		}
		is_at_war = no
		is_part_of_hre = yes
		NOT = {
			num_of_cities = 6
		}
	}
	
	mean_time_to_happen = {
		days = 3
	}
	
	option = {
		name = hee_swabian_events.2.a		# We will join this defensive league.
		ai_chance = {
			factor = 75
		}
		add_country_modifier = {
			name = hee_joined_swabian_league
			duration = -1
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
			}
			country_event = { id = hee_swabian_events.5 days = 30 }
		}
	}
	option = {
		name = hee_swabian_events.2.b		# We stand alone.
		ai_chance = { factor = 25 }
		add_prestige = 5
		add_country_modifier = {
			name = hee_rejected_swabian_league
			duration = -1
			hidden = yes
		}
	}
}

# Thrown out
# THIS IS INTENTIONALLY RANDOM, AND REDUCES LIKELIHOOD THE MORE MEMBERS LIKE YOU
# INCREASES THE MORE MEMBERS DISLIKE YOU

country_event = {
	id = hee_swabian_events.7
	title = hee_swabian_events.7.t
	desc = hee_swabian_events.7.d
	picture = ACCUSATION_eventPicture

	trigger = {
		has_country_modifier = hee_joined_swabian_league
		any_country = {
			has_country_modifier = hee_joined_swabian_league
			defensive_war_with = ROOT
		}
	}

	mean_time_to_happen = {
		days = 600
		modifier = {
			factor = 1.5
			any_country = {
				has_country_modifier = hee_joined_swabian_league
				has_opinion = {
					who = ROOT
					value = 150
				}
			}
		}
		modifier = {
			factor = 0.6
			any_country = {
				has_country_modifier = hee_joined_swabian_league
				NOT = { 	
					has_opinion = {
						who = ROOT
						value = 0
					}
				}
			}
		}
	}

	immediate = {
		remove_country_modifier = hee_joined_swabian_league
		add_country_modifier = { 
			name = hee_rejected_swabian_league
			duration = -1
			hidden = yes
		}
	}

	option = {
		name = hee_swabian_events.7.a
		every_country = {
			limit = {
				has_country_modifier = hee_joined_swabian_league
			}
			reverse_remove_opinion = {
				who = ROOT
				modifier = hee_swabian_league_member
			}
		}
		every_country = {
			limit = {
				has_country_modifier = hee_joined_swabian_league
			}
			country_event = { id = hee_swabian_events.6 days = 3 }
		}
	}
}

# Swabian League disbanded because of lack of membership

	id = hee_swabian_events.8
	title = hee_swabian_events.8.t
	desc = hee_swabian_events.8.d
	picture = CIVIL_WAR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1510
		has_global_flag = hee_swabian_league_established
		NOT = { has_global_flag = hee_swabian_league_disbanded }
		NOT = {
			calc_true_if = {
				all_country = {
					has_country_modifier = hee_joined_swabian_league
				}
				amount = 3
			}
		}
	}

	mean_time_to_happen = {
		months = 24
	}

	option = {
	name = hee_swabian_events.8.a
		every_country = {
			limit = {
				has_country_modifier = hee_joined_swabian_league
			}
			country_event = { id = hee_swabian_events.9 days = 1 }
		}
		every_country = {
			limit = {
				any_neighbor_country = { has_country_modifier = hee_joined_swabian_league }
				NOT = { has_country_modifier = hee_joined_swabian_league }
			}
			country_event = { id = hee_swabian_events.22 days = 7 }
		set_global_flag = hee_swabian_league_disbanded
		}
	}
}

# Request League Support
# Event if a member is fighting a defensive war against a foreign country

country_event = {
	id = hee_swabian_events.12
	title = hee_swabian_events.12.t
	desc = hee_swabian_events.12.d
	picture = BATTLE_eventPicture

	trigger = {
		has_country_modifier = hee_joined_swabian_league
		NOT = { 
			has_global_flag = hee_swabian_league_disbanded 
		}
		any_country = {
			NOT = {
				has_country_modifier = hee_joined_swabian_league
			}
			offensive_war_with = ROOT
		}	
		NOT = { has_country_modifier = hee_league_aid }
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.75
				any_country = {
					war_score_against = {
						who = ROOT
						value = 10
					}
				}
			}
		}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = hee_league_aid
				duration = 3650
			}
		}
	}

	option = {
		name = hee_swabian_events.12.a
		every_country = {
			limit = { 
				has_country_modifier = hee_joined_swabian_league 
			}
			country_event = { id = hee_swabian_events.13 days = 3 }
		}
	}

	option = {
		name = hee_swabian_events.12.b
		add_years_of_income = 0.25
		add_manpower = 0.5
		add_prestige = 1
	}
}

# League Aids Stability in the Region

country_event = {
	id = hee_swabian_events.18
	title = hee_swabian_events.18.t
	desc = hee_swabian_events.18.d
	picture = LAND_MILITARY_eventPicture

	trigger = {
		has_country_modifier = hee_joined_swabian_league
		is_at_war = no
		calc_true_if = {
			all_country = {
				has_country_modifier = hee_joined_swabian_league
				NOT = {
					had_recent_war = 10
				}
			}
			amount = 3
		}
		NOT = {
			has_global_flag = hee_swabian_league_disbanded
		}
	}

	mean_time_to_happen = {
		months = 160
		modifier = {
			factor = 0.75
			any_country = {
				alliance_with = ROOT
				has_country_modifier = hee_joined_swabian_league
			}
		}
		modifier = {
			factor = 0.75
			calc_true_if = {
				all_country = {
					alliance_with = ROOT
					has_country_modifier = hee_joined_swabian_league
				}
				amount = 2
			}
		}
		modifier = {
			factor = 0.75
			calc_true_if = {
				all_country = {
					alliance_with = ROOT
					has_country_modifier = hee_joined_swabian_league
				}
				amount = 3
			}
		}
		modifier = {
			factor = 1.25
			any_country = {
					has_country_modifier = hee_joined_swabian_league
					is_rival = ROOT
			}
		}
	}

	option = {		# And with the league, our country is safe
		name = hee_swabian_events.18.a
		add_stability = 1
	}

	option = {		# This is a golden age for Swabia!
		name = hee_swabian_events.18.b
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
	}

	option = {		# Let's make sure we collect our League taxes.
		name = hee_swabian_events.18.c
		add_treasury = 25
