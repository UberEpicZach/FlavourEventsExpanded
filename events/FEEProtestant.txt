# Protestant Events

namespace = fee_protestant

# Johannes Bugenhagen, second Apostle of the North

country_event = {
	id = fee_protestant.1
	title = fee_protestant.1.T
	desc = fee_protestant.1.D
	picture = ADVISOR_eventPicture

	trigger = {
		OR = {
			capital_scope = { region = scandinavia_region }
			tag = MKL
			tag = POM
			tag = HSA
			tag = EFR
			tag = BRU
			tag = SHL
			tag = BRE
		}
		NOT = { is_year = 1560 }
		NOT = { has_country_modifier = fee_johannes_bugenhagen }
		religion = protestant
		any_province = { religion = catholic }
	}

	mean_time_to_happen = {
		months = 480
		modifier = {
			factor = 0.8
			ADM = 4
		}
		modifier = {
			factor = 0.8
			has_idea_group = humanist_ideas
		}
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = fee_johannes_bugenhagen
				duration = -1
				hidden = yes
			}
		}
	}

	option = {
		name = fee_protestant.1.OPT1
		define_advisor = {
			type = inquisitor
			name = "Johannes Bugenhagen"
			skill = 2
			discount = yes
		}
	}

	option = {
		name = fee_protestant.1.OPT2
		add_stability_or_adm_power = yes
	}
}
