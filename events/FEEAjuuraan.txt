namespace = fee_AJU

#Integrating Mogadishu currency

country_event = {
	id = fee_AJU.1
	title = fee_AJU.1.T
	desc = fee_AJU.1.D
	picture = CORRUPTION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = AJU
		OR = {
			AND = {
				NOT = {
					exists = MDI
				}
				owns_core_province = 1205
			}
			MDI = {
				is_subject_of = AJU
			}
		}
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.5
			has_idea_group = economic_ideas
		}
	}

	option = {
		name = fee_AJU.1.OPT1
		add_country_modifier = {
			name = fee_mogadishu_currency
			duration = -1
		}
		1205 = {
			add_province_modifier = {
				name = fee_mogadishu_administration
				duration = -1
			}
		}
	}

	option = {
		name = fee_AJU.1.OPT2
		add_stability_or_adm_power = yes
	}
}

	id = fee_AJU.2
	title = fee_AJU.2.T
	desc = fee_AJU.2.D
	picture = HARBOUR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = AJU
			tag = MDI
		}
		owns_core_province = 1205
		is_at_war = no
		is_year = 1500
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = fee_AJU.2.OPT1
		ai_chance = {
			factor = 95
		}
		1205 = {
			add_base_production = 1
			add_province_modifier = {
				name = fee_toob_menadir
				duration = -1
			}
		}
		MAM = {
			country_event = { id = fee_AJU.3 }
		}
		SYR = {
			country_event = { id = fee_AJU.3 }
		}
		if = {
			limit = {
				TUR = {
					OR = {
						owns_core_province = 382
						owns_core_province = 361
					}
				}
			}
			country_event = { id = fee_AJU.3 }
		}
	}
	option = {
		name = fee_AJU.2.OPT2
		ai_chance = {
			factor = 5
		}
		add_prestige = 15
	}
}

# Islamic Immigration
country_event = {
	id = fee_AJU.5
	title = fee_AJU.5.T
	desc = fee_AJU.5.D
	picture = RELIGION_eventPicture

	trigger = {
		tag = AJU
		is_year = 1480
		NOT = {
			is_year = 1700
		}
		OR = {
			OR = {
				GRA = {
					OR = {
						religion = sunni
						religion = shiite
						religion = ibadi
					}
					NOT = {
						num_of_cities = 6
					}
				}
				GRA = {
					exists = no
				}
			}
			any_country = {
				capital_scope = {
					region = arabia_region
				}
				OR = {
					religion = sunni
					religion = shiite
					religion = ibadi
				}
				war_exhaustion = 3
			}
		}
	}

	mean_time_to_happen = {
		months = 1200
	}

	option = {
		name = fee_AJU.5.OPT1
		define_advisor = {
			type = theologian
			skill = 2
			discount = yes
		}
		define_advisor = {
			type = diplomat
			skill = 2
			discount = yes
		}
		add_piety = 0.2
		add_prestige = 10
		add_adm_power = 25
	}
	option = {
		name = fee_AJU.5.OPT2
		create_general = { tradition = 40 }
		add_piety = -0.2
		add_prestige = 10
		add_mil_power = 25
	}
	option = {
		name = fee_AJU.5.OPT3
		capital_scope = {
			add_base_production = 1
			add_base_tax = 1
			add_base_manpower = 1
		}
		add_dip_power = 25
		add_prestige = 5
	}
}
