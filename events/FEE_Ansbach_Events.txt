# Dux Francorum Orientalium
country_event = {
	id = FEE_Ansbach_Events.9
	title = FEE_Ansbach_Events.9.T
	desc = FEE_Ansbach_Events.9.D
	picture = ELECTION_REPUBLICAN_eventPicture

	fire_only_once = yes

	trigger = {
		has_country_flag = fee_ansbanch_country_franconian_ambition
		has_ruler_flag = fee_ansbanch_ruler_franconian_ambition
		calc_true_if = {
			all_province = {
				culture = franconian
				OR = {
					owner = {
						dynasty = "von Hohenzollern"
					}
					owner = {
						overlord = {
							dynasty = "von Hohenzollern"
						}
					}
				}
			}
			amount = 5
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	option = {
		name = FEE_Ansbach_Events.9.OPT1
		ai_chance = {
			factor = 1
		}

		add_prestige = 50
		clr_country_flag = fee_ansbanch_country_franconian_ambition
		every_province = {
			limit = {
				culture = franconian
				OR = {
					owner = {
						dynasty = "von Hohenzollern"
					}
					owner = {
						overlord = {
							dynasty = "von Hohenzollern"
						}
					}
				}
			}
			cede_province = ANS
		}
		ANS = {
			franconia_area = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				add_core = ANS
			}
			upper_franconia_area = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
				}
				add_core = ANS
			}
			change_tag = FKN
			restore_country_name_effect = yes
			franconia_area = {
				limit = {
					NOT = { owned_by = root }
					NOT = { is_permanent_claim = root }
				}
				add_permanent_claim = root
			}
			upper_franconia_area = {
				limit = {
					NOT = { owned_by = root }
					NOT = { is_permanent_claim = root }
				}
				add_permanent_claim = root
			}
			add_country_modifier = {
				name = "centralization_modifier"
				duration = 7300
			}
			set_country_flag = formed_franconia_flag
			if = {
				limit = {
					total_development = 100
					OR = {
						is_elector = yes
						is_part_of_hre = no
					}
				}
				set_government_rank = 2
			}
			swap_non_generic_missions = yes
			if = {
				limit = {
					is_playing_custom_nation = no
				}
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
		}
	}
}
