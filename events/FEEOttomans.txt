#Storming the Gates of Europe
country_event = {
	id = fee_TUR.8
	title = fee_TUR.8.T
	desc = fee_TUR.8.D
	picture = COUNTRY_COLLAPSE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = TUR
		OR = { 
			owns = 134
			any_subject_country = { owns = 134 }
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {	#Let us reinforce our European gains.
		name = fee_TUR.8.OPT1
		134 = { 
			add_base_tax = 1
			add_base_manpower = 1
		}
		134 = { 
			add_province_modifier = { 
				name = fee_conquest_wien
				duration = -1
			}
		}
	}
	option = { #Tis but another border province. Let them rebuild.
		name = fee_TUR.8.OPT2
		134 = {
			add_base_production = 1
		}
	}
}

#Murad II's Return to the Throne
country_event = {
	id = fee_TUR.10
	title = fee_TUR.10.T
	desc = fee_TUR.10.D
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = TUR
		has_ruler = "Mehmet II Fatih"
		NOT = { is_year = 1450 }
		NOT = { stability = 2 }
		is_subject = no
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = fee_murad_check
				duration = -1
				hidden = yes
			}
		}
	}
	
	option = {
		name = fee_TUR.10.OPT1
		ai_chance = {
			factor = 100
		}
		hidden_effect = {
			add_country_modifier = {
				name = ruler_deposed
				duration = -1
				hidden = yes
			}
		}
		define_ruler = {
			name = "Murad II"
			dynasty = Osmanoglu
			adm = 4
			dip = 3
			mil = 4
			age = 42
		}
		define_heir = {
			name = "Mehmet II Fatih"
			dynasty = Osmanoglu
			birth_date = 1432.3.30
			adm = 5
			dip = 5
			mil = 6
		}
		add_adm_power = 50
		add_country_modifier = {
			name = fee_janissaries_calmed
			duration = 3650
		}
	}
	option = {
		name = fee_TUR.10.OPT2
		ai_chance = {
			factor = 0
		}
		add_stability = -1
		add_country_modifier = {
			name = fee_janissaries_against_ruler
			duration = 3650
		}
		capital_scope = {
			noble_rebels = 2
		}
	}
}
