#Alhambra
country_event = {
	id = FEE_Monuments_Events.1
	title = FEE_Monuments_Events.1.T
	desc = FEE_Monuments_Events.1.D
	picture = GREAT_BUILDING_eventPicture
	goto = fee_monument_province

	immediate = {
		hidden_effect = {
			223 = { save_event_target_as = fee_monument_province }
		}
	}

	fire_only_once = yes
	trigger = {
		223 = {
			owned_by = ROOT
			has_great_project = {
				type = alhambra
				tier = 3
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = FEE_Monuments_Events.1.OPT1
		ai_chance = {
			factor = 50
		}

		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_splendor = 200
	}
}
