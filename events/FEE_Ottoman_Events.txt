#Storming the Gates of Europe
country_event = {
	id = FEE_Ottoman_Events.2
	title = FEE_Ottoman_Events.2.T
	desc = FEE_Ottoman_Events.2.D
	picture = COUNTRY_COLLAPSE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = TUR
		owns_or_non_sovereign_subject_of = 134
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {	#Let us reinforce our European gains.
		name = FEE_Ottoman_Events.2.OPT1

		134 = {
			add_base_manpower = 1
			add_province_modifier = {
				name = fee_tur_european_stronghold
				duration = -1
			}
		}
	}

	option = { #Tis but another border province. Let them rebuild.
		name = FEE_Ottoman_Events.2.OPT2

		134 = {
			add_base_production = 1
		}
	}
}

#Murad II's Return to the Throne
country_event = {
	id = FEE_Ottoman_Events.3
	title = FEE_Ottoman_Events.3.T
	desc = FEE_Ottoman_Events.3.D
	picture = COURT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = TUR
		has_ruler = "Mehmet II Fatih"
		NOT = { is_year = 1450 }
		NOT = { stability = 2 }
		is_subject = no
	}

	mean_time_to_happen = {
		months = 24
	}

	immediate = {
		hidden_effect = {
			set_country_flag = fee_tur_murad_reign
		}
	}

	option = {
		name = FEE_Ottoman_Events.3.OPT1
		ai_chance = {
			factor = 100
		}

		define_ruler = {
			name = "Murad II"
			dynasty = Osmanoglu
			adm = 4
			dip = 3
			mil = 4
			age = 42
		}
		define_heir = {
			name = "Mehmet II Fatih"
			dynasty = Osmanoglu
			birth_date = 1432.3.30
			adm = 5
			dip = 5
			mil = 6
		}
		add_adm_power = 50
		add_country_modifier = {
			name = fee_tur_jannisariescalmed
			duration = 3650
		}
	}

	option = {
		name = FEE_Ottoman_Events.3.OPT2
		ai_chance = {
			factor = 0
		}

		add_stability = -1
		add_country_modifier = {
			name = fee_tur_jannisariesagainst_ruler
			duration = 3650
		}
		capital_scope = {
			noble_rebels = 2
		}
	}
}
