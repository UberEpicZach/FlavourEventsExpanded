		}

		}

			}

country_event = {
	id = FEE_Western_India_Events.12
	title = FEE_Western_India_Events.12.T
	desc = FEE_Western_India_Events.12.D
	picture = SIEGE_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			export_to_variable = {
				which = fee_chittor_tradition_var
				value = trigger_value:base_manpower
				who = FROM
			}
			export_to_variable = {
				which = fee_chittor_months_var
				value = trigger_value:base_manpower
				who = FROM
			}
		}
	}

	trigger = {
		ROOT = { tag = MUG }
		FROM = {
			province_id = 518
			owner = { primary_culture = rajput }
		}
	}

	option = {
		name = FEE_Western_India_Events.12.OPT1
		custom_tooltip = fee_siege_of_chittor_tt
		while = {
			limit = {
				check_variable = {
					which = fee_chittor_tradition_var
					value = 1
				}
			}
			add_army_tradition_or_mil_power = { amount = 1 }
			change_variable = {
				which = fee_chittor_tradition_var
				value = -1
			}
		}
		while = {
			limit = {
				check_variable = {
					which = fee_chittor_months_var
					value = 1
				}
			}

				which = fee_chittor_months_var
				value = -1
			}
		}
		if = {
			limit = {
				ruler_has_personality = bold_fighter_personality
			}
			add_ruler_modifier = {
				name = fee_conqueror_of_chittor

		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = rajput

					name = fee_mughal_rajput_resistance

			}
		}

			add_base_manpower = -3
			add_province_modifier = {
				name = fee_western_india_assaulted_ramparts
				duration = 9125
			}

				limit = {
					is_expanded_mod_active = { mod = great_monuments }

				}
