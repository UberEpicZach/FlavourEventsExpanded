#Privateer Contracts

country_event = {
	id = fee_barbary.1
	title = fee_barbary.1.t
	desc = fee_barbary.1.d
	picture = SHIP_SAILING_eventPicture

	trigger = {
		capital_scope = {
			region = maghreb_region
		}
		any_owned_province = {
			region = maghreb_region
			has_port = yes
		}
		religion_group = muslim
		is_year = 1490
		NOT = { 
			is_year = 1520
		}
		NOT = {
			has_country_modifier = fee_support_barbary
		}
		NOT = {
			OR = {
				tag = MAM
				tag = EGY
			}
		}
		OR = {
			is_subject = no
			overlord = {
				OR = {
					religion = sunni
					religion = ibadi
					religion = shiite
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.8
			TUR = {
				num_of_cities = 20
			}
		}
	}

	option = {
		name = fee_barbary.1.a
		ai_chance = {
			factor = 80
		}
		add_country_modifier = {
			name = fee_support_barbary
			duration = -1
		}
		add_dip_power = -25
		custom_tooltip = fee_barbary.1.tt
	}

	option = {
		name = fee_barbary.1.b
		ai_chance = {
			factor = 20
		}
		add_stability_or_adm_power = yes
	}
}

#Retaliation
country_event = {
	id = fee_barbary.7
	title = fee_barbary.7.t
	desc = fee_barbary.7.d
	picture = NAVAL_MILITARY_eventPicture

	trigger = {
		has_country_modifier = fee_raided_barbary_country
		NOT = {
			has_country_modifier = fee_barbary_retaliation
		}
		calc_true_if = {
			any_owned_province = {
				has_province_modifier = fee_raided_barbary
			}
			amount = 2
		}
		is_at_war = no
		any_country = {
			has_country_modifier = fee_support_barbary
			NOT = {
				truce_with = ROOT
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = fee_barbary.7.a
		ai_chance = {
			factor = 60
		}
		add_manpower = -1
		add_treasury = -10
		add_mil_power = 25
		add_country_modifier = {
			name = fee_barbary_retaliation
			duration = 5475
			hidden = yes
		}
		random_country = {
			limit = {
				has_country_modifier = fee_support_barbary
				NOT = {
					truce_with = ROOT
				}
			}
			country_event = {
				id = fee_barbary.8 days = 20
			}
		}
	}
	
	option = {
		name = fee_barbary.7.b
		ai_chance = {
			factor = 20
		}
		random_country = {
			limit = {
				has_country_modifier = fee_support_barbary
			}
			reverse_add_casus_belli = {
				target = ROOT
				type = cb_trade_conflict
				months = 120
			}
		}
	}
	option = {
		name = fee_barbary.7.b
		ai_chance = {
			factor = 20
		}
		add_dip_power = 15
	}
}
