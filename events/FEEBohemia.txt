#A Message of Peace
country_event = {
	id = events_boh.3
	title = "events_boh.3.tt"
	desc = "events_boh.3.dd"
	picture = REFORM_eventPicture

	fire_only_once = yes

	trigger = {
		tag = BOH
		has_ruler = "Jirí"
		dynasty = "z Podebrad"
		OR = {
			religion = hussite
			ruler_religion = hussite
		}
 	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = events_boh.3.a #Unfortunate, but we can make some good out of this.
		if = {
			limit = {
				exists = TUR
			}
			every_country = {
				limit = {
					is_rival = TUR
					religion_group = christian
				}
				add_opinion = {
					who = BOH
					modifier = boh_charter_turk_rivals
				}
			}
			TUR = {
				add_opinion = {
					who = BOH
					modifier = boh_charter_turks
				}
			}
		}
		if = {
			limit = {
				exists = FRA
				NOT = {
					FRA = {
						is_rival = TUR
					}
				}
			}
			FRA = {
				add_opinion = {
					who = BOH
					modifier = boh_charter_france
				}
			}
		}
		define_advisor = {
			type = statesman
			name = "Zdenek Lev of Rozmital"
			skill = 2
			culture = czech
			religion = catholic
			discount = yes
		}
		add_prestige = 15
	}
	option = {
		name = events_boh.3.b #How naïve! Perhaps we should forget the whole thing.
		add_prestige = -10
		add_dip_power = 50
	}
}
