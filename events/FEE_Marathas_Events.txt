	title = FEE_Marathas_Events.1.T
	desc = FEE_Marathas_Events.1.D

	trigger = {

			num_of_cities = 10
		}
		NOT = {
			any_country = { has_country_flag = fee_marathas_marathi_enemy }
		}
		NOT = { primary_culture = marathi }
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 5
			religion = hinduism
		}
		modifier = {
			factor = 1.5
			accepted_culture = marathi
		}
		modifier = {
			factor = 0.8
			religion = sunni
		}
		modifier = {
			factor = 1.5
			ai = no
			is_at_war = yes
		}
		modifier = {
			factor = 0.8
			NOT = {
				num_of_cities = 11
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = fee_marathas_marathi_enemy
			random_owned_province = {
				limit = {
					culture = marathi
					religion = hinduism
				}
				add_core = MAR
			}
			release = MAR
			MAR = {
				define_ruler = {
					name = "Shivaji"
					dynasty = "Bhonsle"
					ADM = 4
					DIP = 4
					MIL = 5
				}
			}
			every_province = {
				limit = {
					culture = marathi
				}
				add_claim = MAR
				add_core = MAR
			}
		}
	}

	option = {
		name = FEE_Marathas_Events.1.OPT1

		hidden_effect = {
			MAR = {
				add_manpower = 20
				add_treasury = 100
				add_adm_power = 100
				add_dip_power = 100
				add_mil_power = 100
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
			}
		}
		MAR = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
}
