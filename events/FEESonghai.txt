		tag = SON

			is_year = 1490
		}

	}

		months = 100
	}

			name = "Muhammad Askia"

		}

			add_country_modifier = {

				hidden = yes
			}
		}
	}
}

#The Heretic Heir
country_event = {
	id = fee_SON.2
	title = fee_SON.2.t
	desc = fee_SON.2.d
	picture = RELIGIOUS_CONVERSION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = SON
		NOT = {
			is_year = 1510
		}
		is_year = 1485
		religion = sunni
		has_leader = "Muhammad Askia"
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.6
			has_country_flag = fee_general_ture
		}
		modifier = {
			factor = 1.3
			ADM = 5
		}
		modifier = {
			factor = 1.2
			DIP = 5
		}
		modifier = {
			factor = 1.2
			MIL = 5
		}
	}

	option = {
		name = fee_SON.2.a
		ai_chance = {
			factor = 85
		}
		add_country_modifier = {
			name = ruler_deposed
			duration = -1
			hidden = yes
		}
		define_leader_to_ruler = {
			type = general
			name = "Askia Muhammad"
			adm = 6
			dip = 5
			mil = 4
			age = 35
			claim = 60
			fixed = yes
		}
		kill_heir = yes
		add_piety = 0.15
		add_stability = -1
		random_owned_province = {
			spawn_rebels = {
				type = pretender_rebels
				size = 2
				leader = "Baru Sonni"
			}
		}
		random_owned_province = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
			}
		}
	}
	option = {
		name = fee_SON.2.b
		ai_chance = {
			factor = 15
		}
		kill_ruler = yes
		if = {
			limit = {
				has_heir = no
			}
			define_ruler = {
				name = "Baru"
				dynasty = "Sonni"
				age = 15
			}
		}
		change_religion = animism
		add_stability = -2
	}
}
