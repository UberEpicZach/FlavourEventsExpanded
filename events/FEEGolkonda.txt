#Golkondese Events

namespace = fee_GOC

#Expand Golkonda Fort
country_event = {
	id = fee_GOC.1
	title = fee_GOC.1.T
	desc = fee_GOC.1.D
	picture = FORT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = GOC
		is_year = 1507
		NOT = {
			is_year = 1590
		}
		NOT = {
			has_country_modifier = fee_moved_capital
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = fee_GOC.1.OPT1
		542 = {
			add_province_modifier = {
				name = fee_golkonda_fort
				duration = 1
			}
			add_base_manpower = 1
		}
		add_treasury = -25
	}
	
	option = {
		name = fee_GOC.1.OPT2
		add_mil_power = 15
	}
}
