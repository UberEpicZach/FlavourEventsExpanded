#The revolt of Adolf van Egmond

	id = fee_NED.3
	title = fee_NED.3.t
	desc = fee_NED.3.d
	picture = BATTLE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = GEL
		NOT = {
			is_year = 1480 
		}
		exists = BUR
		NOT = {
			num_of_cities = 4
		}
		owns = 99
		is_year = 1450
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			NOT = {
				legitimacy = 80
			}
		}
	}
	
	option = {
		name = fee_NED.3.a
		random_owned_province = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
				leader = "Adolf van Egmond"
			}
		}
		hidden_effect = {
			add_country_modifier = {
				name = fee_adolf_revolted
				duration = 3650
				hidden = yes
			}
		}
	}
	
	option = {
		name = fee_NED.3.b
		add_dip_power = -150
	}
}
