# The Field of the Cloth of Gold
country_event = {
	id = FEE_England_Great_Britain_Events.19
	title = FEE_England_Great_Britain_Events.19.T
	desc = FEE_England_Great_Britain_Events.19.D
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = ENG
			tag = GBR
		}
		OR = {
			AND = {
				any_country = {
					dynasty = "von Habsburg"
					is_rival = ROOT
				}
			}
			SPA = {
				is_rival = ROOT
			}
		}
		ROOT = {
			is_part_of_hre = no
			NOT = { dynasty = "von Habsburg" }
			NOT = { junior_union_with = SPA }
		}
		FRA = {
			is_part_of_hre = no
			NOT = { dynasty = "von Habsburg" }
			NOT = { junior_union_with = SPA }
		}
		SPA = {
			OR = {
				NOT = {	junior_union_with = ROOT }
				NOT = { junior_union_with = FRA }
			}
		}
		is_year = 1490
		NOT = { is_year = 1560 }
	}

	option = {
		name = FEE_England_Great_Britain_Events.19.OPT1

		add_years_of_income = 0.2
		if = {
			limit = {
				FRA = {
					historical_rival_with = ROOT
				}
			}
			FRA = {
				remove_historical_rival = ROOT
			}
			remove_historical_rival = FRA
		}
		else = {
			FRA = {
				add_opinion = {
					who = ROOT
					modifier = fee_eng_gbr_field_of_the_cloth_of_gold
				}
			}
			add_opinion = {
				who = FRA
				modifier = fee_eng_gbr_field_of_the_cloth_of_gold
			}
		}
	}

	option = {
		name = FEE_England_Great_Britain_Events.19.OPT2

		FRA = {
			add_opinion = {
				who = ROOT
				modifier = fee_eng_gbr_rejected_alliance
			}
		}
		add_prestige = -10
	}
}
