				}
			}
			AND = {
				tag = ARA
				NOT = {

				}
			}
		}
		owns = 210
		is_year = 1570
		NOT = {
			is_year = 1630
		}
		exists = FRA
		NOT = {
			alliance_with = FRA
		}
		NOT = {
			has_opinion = {

				value = 50
			}
		}
	}

	mean_time_to_happen = {
		months = 420
		modifier = {
			factor = 0.7
			truce_with = FRA
		}
		modifier = {
			factor = 0.8
			is_rival = FRA
		}
		modifier = {
			factor = 0.8
			FRA = {
				is_rival = ROOT
			}
		}
	}

	option = { #Build a great fort!
		name = fee_NAV.11.OPT1
