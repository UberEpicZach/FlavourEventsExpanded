country_event = {

	fire_only_once = yes
	
	trigger = {
		has_country_flag = CE_franconian_ambition
		has_ruler_flag = CE_franconian_ambition

			all_province = {
				culture = franconian
				OR = {
					owner = {
						dynasty = "von Hohenzollern"
					}
					owner = {
						overlord = {
							dynasty = "von Hohenzollern"
						}
					}
				}
			}
			amount = 5
		}
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	option = {
		name = CE_ansbach_expanded.9.OPT1
		clr_country_flag = CE_franconian_ambition
		every_province = {
			limit = {
				culture = franconian
				OR = {
					owner = {
						dynasty = "von Hohenzollern"
					}
					owner = {
						overlord = {
							dynasty = "von Hohenzollern"
						}
					}
				}
			}
			cede_province = ANS 
		}
		ANS = {
			restore_country_name_effect = yes
			franconia_area = {
				limit = {
					NOT = { owned_by = ANS }
					NOT = { is_permanent_claim = ANS }
				}
				add_permanent_claim = ANS
			}
			upper_franconia_area = {
				limit = {
					NOT = { owned_by = ANS }
					NOT = { is_permanent_claim = ANS }
				}
				add_permanent_claim = ANS
			}
			every_owned_province = {
				limit = {
					OR = {
						area = franconia_area
						area = upper_franconia_area
					}
				}
				add_core = ANS
			}
			change_tag = FKN
			set_country_flag = formed_franconia_flag
			if = {
				limit = {
					is_playing_custom_nation = no
				}
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
		}
		add_prestige = 50
		ai_chance = {
			factor = 1
		}
	}
}
