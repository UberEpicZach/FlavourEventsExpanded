#Saint Olav's Shrine for Overlord
country_event = {
	id = FEE_Norway_Events.2
	title = FEE_Norway_Events.2.T
	desc = FEE_Norway_Events.2.D
	picture = IMPORTANT_STATUE_eventPicture

	fire_only_once = yes

	trigger = {
		overlord_of = NOR
		OR = {
			religion = protestant
			religion = reformed
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5
			OR = {
				NOR = {
					religion = protestant
				}
				NOR = {
					religion = reformed
				}
			}
		}
	}

	option = {
		name = FEE_Norway_Events.2.OPT1
		ai_chance = {
			factor = 90
		}

		NOR = {
			country_event = {
				id = FEE_Norway_Events.3 days = 10 random = 5
			}
			add_treasury = 25
			random_owned_province = {
				limit = {
					NOT = { has_owner_religion = yes }
				}
				change_religion = ROOT
			}
		}
	}

	option = {
		name = FEE_Norway_Events.2.OPT2
		ai_chance = {
			factor = 10
		}

		add_prestige = -5
	}
}
