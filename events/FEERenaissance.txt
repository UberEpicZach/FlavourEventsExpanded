		OR = {
			tag = LAN
			tag = LUC
			tag = MLO
			tag = VEN
			tag = PAP
			tag = URB
			tag = FER
			tag = MOD
			tag = GEN
			tag = MAN
			tag = MFA
			tag = PAR
			tag = PIS
			tag = TUS
			tag = AQU
			tag = ETR
		}
		NOT = { has_country_modifier = fee_renaissance }
		NOT = { is_year = 1610 }
		NOT = {
			has_country_modifier = fee_had_renaissance
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = fee_renaissance.1.a

			name = fee_renaissance

		}

# End of renaissance influences
country_event = {
	id = fee_renaissance.3
	title = fee_renaissance.3.t
	desc = fee_renaissance.3.d
	picture = BIG_BOOK_eventPicture

	trigger = {
		has_country_modifier = fee_renaissance
		is_year = 1580
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.5
			NOT = {
				any_neighbor_country = {
					has_country_modifier = fee_renaissance
				}
			}
		}
		modifier = {
			factor = 0.2
			is_year = 1610
		}
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = fee_had_renaissance
				duration = -1
				hidden = yes
			}
		}
	}

	option = {
		name = fee_renaissance.3.a
		remove_country_modifier = fee_renaissance
	}
}

# Papal Legate Concedes to Humanist
country_event = {
	id = fee_renaissance.12
	title = fee_renaissance.12.t
	desc = fee_renaissance.12.d
	picture = RELIGION_eventPicture
	
	trigger = {
		ai = no
		has_country_modifier = fee_renaissance
		NOT = {
			is_religion_enabled = protestant
		}
		religion = catholic
	}
	
	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 0.8
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 0.8
			reverse_has_opinion = {
				who = PAP
				value = 100
			}
		}
	}
	
	option = {
		name = fee_renaissance.12.a
		add_prestige = 10
		add_reform_desire = -0.01
	}
	
	option = {
		name = fee_renaissance.12.b
		define_advisor = {
			type = philosopher
			skill = 2
			discount = yes
		}
	}
}
