#The Diet of Worms
country_event = {
	id = FEE_Austria_Events.1
	title = FEE_Austria_Events.1.T
	desc = FEE_Austria_Events.1.D
	picture = HRE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = HAB
		is_emperor = yes
		is_year = 1480
		NOT = {
			is_religion_enabled = protestant
		}
		NOT = { hre_reform_passed = hofgericht }
		NOT = { hre_reform_passed = emperor_reichsstabilitaet }
		imperial_influence = 20
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = FEE_Austria_Events.1.OPT1

		add_imperial_influence = 20
		add_dip_power = 20
		add_mil_power = 20
		add_stability = -1
		if = {
			limit = {
				61 = {
					is_part_of_hre = yes
				}
			}
			61 = {
				add_base_tax = 1
			}
		}
		add_country_modifier = {
			name = fee_hab_reforms_diet_worms
			duration = 365
		}
	}

	option = {
		name = FEE_Austria_Events.1.OPT2

		add_imperial_influence = -20
		every_country = {
			limit = {
				is_part_of_hre = yes
			}
			add_opinion = {
				who = HAB
				modifier = fee_hab_protected_imperial_status_quo
			}
		}
	}
}

#Rise of the Ottomans in Europe
country_event = {
	id = FEE_Austria_Events.2
	title = FEE_Austria_Events.2.T
	desc = FEE_Austria_Events.2.D
	picture = HRE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = HAB
		TUR = {
			num_of_cities = 12
		}
		NOT = {
			any_owned_province = {
				NOT = {
					continent = europe
				}
			}
		}
		OR = {
			any_owned_province = {
				any_neighbor_province = {
					owned_by = TUR
				}
			}
			any_subject_country = {
				any_owned_province = {
					any_neighbor_province = {
						owned_by = TUR
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}

	option = {
		name = FEE_Austria_Events.2.OPT1
		ai_chance = {
			factor = 100
		}

		add_mil_power = 50
		add_manpower = 3
		add_army_tradition = 5.0
		add_opinion = {
			who = TUR
			modifier = fee_hab_bulwark_christendom
		}
		reverse_add_opinion = {
			who = TUR
			modifier = fee_hab_bulwark_christendom
		}
	}
	option = {
		name = FEE_Austria_Events.2.OPT2
		ai_chance = {
			factor = 0
		}

		add_dip_power = 25
		add_treasury = 10
	}
}

		}
	}

			factor = 85
		}

		add_country_modifier = {
			name = fee_hab_austrian_baroque

		}

	}
	option = {
		name = FEE_Austria_Events.3.OPT2
		ai_chance = {
			factor = 15
		}

		add_stability_or_adm_power = yes
		add_adm_power = 10
		add_dip_power = 10
	}
}
