country_event = {
	id = FEE_Italian_Wars_Events.102
	title = FEE_Italian_Wars_Events.102.T
	desc = FEE_Italian_Wars_Events.102.D
	picture = BATTLE_eventPicture

	fire_only_once = yes

	trigger = {
		is_year = 1494
		NOT = { is_year = 1498 }
		FRA = {
			has_country_flag = fee_first_italian_war_flag
			offensive_war_with = NAP
		}
		OR = {
			tag = NAP
			tag = FRA
		}
		105 = {
			NOT = { controlled_by = owner }
			controller = {
				OR = {
					tag = FRA
					tag = NAP
					AND = {
						has_country_flag = fee_first_italian_war_flag
						offensive_war_with = NAP
					}
					AND = {
						is_in_war = {
							attacker_leader = FRA
							defender_leader = NAP
						}
						OR = {
							alliance_with = FRA
							overlord = {
								OR = {
									alliance_with = FRA
									AND = {
										offensive_war_with = NAP
										has_country_flag = fee_first_italian_war_flag
									}
								}
							}
						}
					}
					AND = {
						has_country_flag = fee_first_italian_war_flag
						defensive_war_with = FRA
					}
					AND = {
						is_in_war = {
							attacker_leader = FRA
							defender_leader = NAP
						}
						OR = {
							alliance_with = NAP
							overlord = {
								OR = {
									alliance_with = NAP
									AND = {
										defensive_war_with = FRA
										has_country_flag = fee_first_italian_war_flag
									}
								}
							}
						}
					}

			}
		}

	major = yes

		is_in_war = {
			attacker_leader = FRA
			defender_leader = NAP
			casus_belli = cb_restore_personal_union
		}
	}

	option = {
		name = FEE_Italian_Wars_Events.102.OPT1

		every_country = {
			limit = {
				OR = {
					tag = FRA
					AND = {
						has_country_flag = fee_first_italian_war_flag
						offensive_war_with = NAP
					}
					AND = {
						is_in_war = {
							attacker_leader = FRA
							defender_leader = NAP
						}
						OR = {
							alliance_with = FRA
							overlord = {
								OR = {
									alliance_with = FRA
									AND = {
										offensive_war_with = NAP
										has_country_flag = fee_first_italian_war_flag
									}
								}
							}
						}
					}
				}
			}
			add_prestige = -10
			add_army_tradition = 5
			add_war_exhaustion = 1
			add_years_of_income = -0.332
		}
		every_country = {
			limit = {
				OR = {
					tag = NAP
					AND = {
						defensive_war_with = FRA
						has_country_flag = fee_first_italian_war_flag
					}
					AND = {
						is_in_war = {
							attacker_leader = FRA

						}
