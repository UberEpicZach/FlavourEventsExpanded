# Scottish events

namespace = fee_SCO

# The Education Act
country_event = {
	id = fee_SCO.1
	title = fee_SCO.1.T
	desc = fee_SCO.1.D
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = SCO
		is_year = 1480
		NOT = {
			is_year = 1520
		}
		is_subject = no
		government = monarchy
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.8
			ADM = 4
		}
		modifier = {
			factor = 0.8
			advisor = statesman
		}
	}

	# Every noble should be a skilled administrator
	option = {
		name = fee_SCO.1.OPT1
		add_adm_power = -10
		add_country_modifier = {
			name = fee_education_act
			duration = 18250
		}
	}

	# It is better if they focus on military prowess.
	option = {
		name = fee_SCO.1.OPT2
		add_mil_power = 30
	}
}

# Anti-French sentiment

	title = fee_SCO.2.T
	desc = fee_SCO.2.D
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
		tag = SCO
		alliance_with = FRA
		is_at_war = no
		OR = {
			religion = protestant
			religion = reformed
		}
		FRA = {
			religion = catholic
		}
		NOT = {
			is_year = 1580
		}
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.75
			FRA = {
				is_defender_of_faith = yes
			}
		}
	}

	option = {
		name = fee_SCO.2.OPT1
		add_country_modifier = {
			name = fee_scottish_french_supression
			duration = 3650
		}
	}
	option = {
		name = fee_SCO.2.OPT2
		reverse_add_opinion = {
			who = FRA
			modifier = fee_scottish_anti_french
		}
	}
}


# Claim of Rights

country_event = {
	id = fee_SCO.3
	title = fee_SCO.3.T
	desc = fee_SCO.3.D
	picture = GOOD_WITH_MONARCH_eventPicture

	fire_only_once = yes

	trigger = {
		is_lesser_in_union = yes
		OR = {
			tag = SCO
			AND = {
				tag = GBR
				any_owned_province = {
					OR = {
						area = highlands_area
						area = lowlands_area
					}
				}
			}
		}
		is_year = 1600
	}

	mean_time_to_happen = {
		months = 2
	}

	option = {
		name = fee_SCO.3.OPT1
		add_country_modifier = {
			name = fee_claim_suppressed
			duration = 7200
		}
		overlord = {
			add_opinion = {
				who = ROOT
				modifier = fee_subject_claim_supressed
			}
		}
	}
	option = {
		name = fee_SCO.3.OPT2
		overlord = {
			country_event = {
				id = fee_SCO.4 days = 5 tooltip = fee_SCO.3.OPT2.tt
			}
		}
	}
}

# Scotland Presents its Claim of Right

	desc = fee_SCO.4.D
	picture = CONQUEST_eventPicture

	is_triggered_only = yes

	trigger = {
		SCO = {
			is_subject_of = ROOT
		}
	}

		name = fee_SCO.4.OPT1
		FROM = {
			add_country_modifier = {
				name = fee_claim_supressed
				duration = 10000
			}
			add_liberty_desire = 25
		}
	}
	option = {
		name = fee_SCO.4.OPT2
		FROM = {
			every_province = {
				limit = {
					OR = {
						area = lowlands_area
						area = highlands_area
					}
				}
			add_local_autonomy = 50
			}
		}
	}
}
