#Payment from tributary
country_event = {
	id = hee_MUG.3
	title = hee_MUG.3.t
	desc = hee_MUG.3.d
	picture = CONQUEST_eventPicture
	
	trigger = {
		tag = MUG
		NOT = {
			has_country_modifier = hee_MUG_received_tribute
		}
		any_subject_country = {
			NOT = {
				has_country_modifier = hee_paid_MUG_tribute
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = hee_MUG_received_tribute
				duration = 1825
				hidden = yes
			}
		}
	}
	
	option = {
		name = hee_MUG.3.a
		add_years_of_income = 0.2
		random_subject_country = {
			reverse_add_opinion = {
				who = MUG
				modifier = hee_MUG_tribute
			}
			add_country_modifier = {
				name = hee_paid_MUG_tribute
				duration = 3650
				hidden = yes
			}
		}
	}
}
