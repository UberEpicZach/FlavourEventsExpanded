# Kronstadt Fortress.
country_event = {
	id = FEE_Russia_Events.1
	title = FEE_Russia_Events.1.T
	desc = FEE_Russia_Events.1.D
	picture = CITY_VIEW_eventPicture
	fire_only_once = yes

	mean_time_to_happen = {
		months = 300
	}

	trigger = {
		tag = RUS
		has_country_flag = relocated_capital_st_petersburg
		owns = 295
		owns = 33
		33 = {
			is_capital = yes
		}
	}

	option = {
		name = FEE_Russia_Events.1.OPT1
		ai_chance = { factor = 70 }

		add_years_of_income = -0.75
		33 = {
			# Grand Shipyard check.
			if = {
				limit = {
					NOT = { has_shipyard_building_trigger = yes }
				}
				add_building = grand_shipyard
			}
			else_if = {
				limit = {
					NOT = {
						has_building = grand_shipyard
					}
					has_building = shipyard
				}
				remove_building = shipyard
				add_building = grand_shipyard
			}
			else = {
				add_base_production = 1
			}
			# Naval Battery check.
			if = {
				limit = {
					NOT = { has_building = coastal_defence }
					NOT = { has_building = naval_battery }
				}
				add_building = naval_battery
			}
			else_if = {
				limit = {
					NOT = {
						has_building = naval_battery
					}
					has_building = coastal_defence
				}
				remove_building = coastal_defence
				add_building = naval_battery
			}
			else = {
				add_base_tax = 1
			}
			# Fort check.
			if = {
				limit = {
					NOT = { has_building = fort_15th }
				}
				add_building = fort_15th
			}
			else_if = {
				limit = {
					NOT = { has_building = fort_16th }
				}
				add_building = fort_16th
			}
			else_if = {
				limit = {
					NOT = { has_building = fort_17th }
				}
				add_building = fort_17th
			}
			else_if = {
				limit = {
					NOT = { has_building = fort_18th }
				}
				add_building = fort_18th
			}
			else = {
				add_base_manpower = 1
			}
			# Kronstadt Fortress modifier
			add_province_modifier = {
				name = fee_rus_kronstadt_fortress
				duration = -1
			}
		}
	}

	option = {
		name = FEE_Russia_Events.1.OPT2
		ai_chance = { factor = 30 }

		add_years_of_income = -0.5
		# Renovated Kremlin modifier
		295 = {
			add_province_modifier = {
				name = fee_rus_renovated_kremlin
				duration = -1
			}
		}
	}
}
