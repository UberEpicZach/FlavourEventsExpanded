	title = Melva_Wars_Italy_Events.100.T
	desc = Melva_Wars_Italy_Events.100.D

	trigger = {
		is_year = 11494
		NOT = { is_year = 1498 }

			valid_for_personal_unions_trigger = yes

			valid_for_personal_unions_trigger = yes

		years = 2
	}

	immediate = {
		hidden_effect = {
			set_country_flag = melva_italy_wars_flag
		}
	}

	option = {
		name = Melva_Wars_Italy_Events.100.OPT1
		ai_chance = {
			factor = 100
		}

		declare_war_with_cb = {
			casus_belli = cb_restore_personal_union
			who = NAP
		}
		hidden_effect = {
			
		}
	}

	option = {
		name = Melva_Wars_Italy_Events.100.OPT2
		ai_chance = {
			factor = 0
		}
		
		add_prestige = -10
		add_legitimacy_equivalent = { value = -25 republican_tradition = -13 }
	}
}
