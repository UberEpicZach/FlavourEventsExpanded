#A Burgundian feast?
country_event = {
	id = fee_BUR.2
	title = fee_BUR.2.t
	desc = fee_BUR.2.d
	picture = FEAST_eventPicture

	fire_only_once = no

	trigger = {
		tag = BUR
		NOT = { is_year = 1480 }
	}

	mean_time_to_happen = {
		months = 180
	}

	# Option A: Host a grand feast!
    option = {
        name = fee_BUR.2.a
		ai_chance = {
			factor = 50
		}
        add_years_of_income = -0.2
        add_prestige = 20
        if = {
            limit = { has_dlc = "Mandate of Heaven" }
            add_splendor = 100
        }
        # Two ifs instead of if-else, ugly, but seen on vanilla in janissary_decadence.txt
        if = { 
            limit = {
                NOT = { has_dlc = "Mandate of Heaven" }
                add_dip_power = 50
            }
        }
	}	
	
	# Option B: We can wait...
    option = {
        name = fee_BUR.2.b
		ai_chance = {
			factor = 50
		}
		add_prestige = -5
	}
}

#Mary of Guelders
country_event = {
	id = fee_BUR.3
	title = fee_BUR.3.t
	desc = fee_BUR.3.d
	picture = MARY_OF_GUELDERS_eventPicture

    trigger = {
        OR = {
            tag = BUR
            tag = GEL
            tag = SCO
        }
        is_year = 1451 
    }
    mean_time_to_happen = {
		months = 240
        modifier = {
		    factor = 0.5
		    tag = BUR
	    }
	}
	fire_only_once = yes
	
    # Option A: She should be an advisor!
    option = {
        name = fee_BUR.3.a
		ai_chance = {
			factor = 50
		}
        # Gains level 2 female stateman advisor Maria van Gelre of Dutch culture and state religion
		define_advisor = {
			type = statesman
            culture = dutch
			religion = root
            skill = 2
            name = "Maria van Gelre"
            female = yes
		}

	}	
	
	# Option B: We can learn from her
    option = {
        name = fee_BUR.3.b
		ai_chance = {
			factor = 30
		}
		add_dip_power = 50
	}

    # Option C: How nice
    option = {
        name = fee_BUR.3.c
		ai_chance = {
			factor = 20
		}
		add_prestige = 10
	}
}
