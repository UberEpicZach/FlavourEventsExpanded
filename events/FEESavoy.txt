# The Army of Savoy

country_event = {
	id = fee_SAV.2
	title = fee_SAV.2.T
	desc = fee_SAV.2.D
	picture = MILITARY_CAMP_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SAV
		NOT = {
			is_year = 1600
		}
		is_year = 1560
		is_subject = no
		OR = {
			mil = 3
			advisor = army_reformer
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			mil = 4
		}
		modifier = {
			factor = 0.9
			NOT = {
				num_of_mercenaries = 1
			}
		}
		modifier = {
			factor = 0.8
			has_idea_group = administrative_ideas
		}
	}
	
	option = {		# We need a standing army.
		name = fee_SAV.2.OPT1
		add_country_modifier = {
			name = fee_savoy_army
			duration = 10950
		}
	}

	option = {		#mercenaries are our greatest strength
		name = fee_SAV.2.OPT2
		add_country_modifier = {
			name = fee_savoy_mercenaries
			duration = 10950
		}
	}
}

	}

		tag = SAV
		is_year = 1600
		NOT = {

		}
	}

		modifier = {

		}

			OR = {
				alliance_with = FRA

			}
		}
	}
	
	option = {
		name = fee_SAV.5.OPT1
		add_adm_power = 25
		add_dip_power = 50
		add_mil_power = -25
	}	
}
