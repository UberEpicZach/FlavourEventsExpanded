# The Qizilbash
country_event = {
	id = fee_PER.1
	title = fee_PER.1.T
	desc = fee_PER.1.D
	picture = HORDE_ON_HORSEBACK_eventPicture

	fire_only_once = yes

	trigger = {
		tag = PER
		NOT = { is_year = 1600 }
		religion = shiite
	}

	mean_time_to_happen = {
		months = 4
	}

	option = { #They are here for us now, that is all that matters.
		name = fee_PER.1.OPT1

		hidden_effect = {
			add_country_modifier = {
				name = per_the_qizilbash
				duration = -1
				hidden = yes
			}
		}
		add_mil_power = 50
	}
}

# The Qizilbash Fight for Power

	desc = fee_PER.3.D
	picture = HORDE_ON_HORSEBACK_eventPicture

	trigger = {
		tag = PER
		has_country_modifier = per_the_qizilbash

		NOT = {	has_ruler_modifier = per_qizilbash_fight }
	}

		months = 6
	}

		name = fee_PER.3.OPT1

				name = per_qizilbash_fight

			}

		add_dip_power = -100
	}
}

# The Third Force	
country_event = {
	id = fee_PER.6
	title = fee_PER.6.T
	desc = fee_PER.6.D
	picture = FARMING_eventPicture

	fire_only_once = yes

	trigger = {
		tag = PER
		NOT = { has_country_modifier = per_end_qizilbash_influence }
		has_country_modifier = per_the_qizilbash	
		is_year = 1580
		mil = 3
		dip = 3
		adm = 3
		caucasia_region = {
			owned_by = PER
		}
	}

	mean_time_to_happen = {
		months = 120
	}

	option = { # We need them to reform our society!
		name = fee_PER.6.OPT1

		random_owned_province = {
			limit = {
				NOT = { base_tax = 4 }
			}
			change_culture = armenian
			add_base_tax = 1
		}
		add_manpower = 15
		add_country_modifier = {
			name = per_the_third_force
			duration = -1
			hidden = yes
			}
		}

	option = { # The Qizibash are no threat to the Shah.
		name = fee_PER.6.OPT2

		add_stability_or_adm_power = yes
	}
}

# End of Tribal Independence
country_event = {
	id = fee_PER.7
	title = fee_PER.7.T
	desc = fee_PER.7.D
	picture = GOOD_WITH_MONARCH_eventPicture

	trigger = {
		tag = PER
		NOT = { has_country_modifier = per_end_qizilbash_influence }
		has_country_modifier = per_the_qizilbash
		has_country_modifier = per_the_third_force
	}

	mean_time_to_happen = {
		months = 180
	}

	option = { # No one will ever hold Persia hostage again!
		name = fee_PER.7.OPT1

		hidden_effect = {
			add_country_modifier = {
				name = per_end_qizilbash_influence
				duration = -1
				hidden = yes
			}
		}
		add_prestige = 20
		add_stability_or_adm_power = yes
		capital_scope = {
			add_base_tax = 1
		}
		hidden_effect = {
			remove_country_modifier = per_the_qizilbash
		}
	}
}
