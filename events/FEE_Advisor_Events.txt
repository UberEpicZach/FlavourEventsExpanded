# A Monumental Reveal (II)
country_event = {
	id = FEE_Advisor_Event.4
	title = FEE_Advisor_Event.EVTNAME4
	desc = FEE_Advisor_Event.EVTDESC4
	picture = CITY_VIEW

	is_triggered_only = yes

	trigger = {
		has_country_flag = fee_events_for_monumental_suggestion_active
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					religion = catholic
					exists = PAP
				}
				PAP = {
					save_event_target_as = fee_statue_target
				}
			}
			else_if = {
				limit = {
					religion = catholic
					any_country = {
						is_papal_controller = yes
					}
				}
				random_country = {
					limit = {
						is_papal_controller = yes
					}
					save_event_target_as = fee_statue_target
				}
			}
			else_if = {
				limit = {
					any_country = {
						religion = ROOT
					}
				}
				random_country = {
					limit = {
						religion = ROOT
					}
					save_event_target_as = fee_statue_target
				}
			}
			else = {
				save_event_target_as = fee_statue_target
			}
		}
	}

	option = {
		name = FEE_Advisor_Event4.EVTOPT1

		if = {
			limit = {
				event_target:fee_statue_target = {
					NOT = { tag = ROOT }
				}
			}
			if = {
				limit = {
					event_target:fee_statue_target = {
						tag = PAP
					}
				}
				every_country = {
					limit = {
						religion = ROOT
					}
					add_opinion = {
						who = ROOT
						modifier = fee_religious_statue
					}
				}
			}
			else = {
				event_target:fee_statue_target = {
					add_opinion = {
						who = ROOT
						modifier = fee_religious_statue
					}
				}
			}
		}
		else = {
			add_prestige = 5
		}
	}

	after = {
		clr_country_flag = fee_events_for_monumental_suggestion_active
	}
}
