country_event = {
	id = disaster_fee_crisis_of_the_mughal_empire.11
	title = disaster_fee_crisis_of_the_mughal_empire.11.T
	desc = disaster_fee_crisis_of_the_mughal_empire.11.D
	picture = REVOLUTION_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		NOT = { exists = RJP }
		num_of_owned_provinces_with = {
			value = 10
			culture = rajput
			is_capital = no
			controlled_by = REB
		}
	}

	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = rajput
					is_capital = no
					controlled_by = REB
				}
				add_core = RJP
				clear_rebels = yes
			}
			capital_scope = {
				if = {
					limit = {
						is_core = RJP
					}
					remove_core = RJP
				}
			}
			release = RJP
			vassalize = RJP
			RJP = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7
						artillery = 0.3
					}
				}
				add_treasury = 1000
				add_manpower = 2
				add_years_of_income = 15
				add_yearly_manpower = 15
				change_government = monarchy
				adopt_reform_progress = ROOT
				capital_scope = {
					PREV = { change_religion = PREV }
				}
			}
		}
	}

	option = {
		name = disaster_fee_crisis_of_the_mughal_empire.11.OPT1

		add_stability = -1
		RJP = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
}
