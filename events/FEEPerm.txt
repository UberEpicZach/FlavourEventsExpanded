#Perm Freed
country_event = {
	id = hee_prm.1
	title = hee_prm.1.t
	desc = hee_prm.1.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
    
	trigger = {
		tag = PRM
		is_subject_of = MOS
		is_year = 1444
		NOT = {
			is_year = 1472
		}
		NOT = {
			has_country_modifier = hee_perm_freed
		}
		OR = {
			ai = yes
			MOS = {
				ai = yes
			}
		}
	}

    mean_time_to_happen = {
		months = 1 
	}

    option = {
	name = hee_prm.1.a
		MOS = {
			free_vassal = PRM
		}
		hidden_effect = {
			add_country_modifier = {
				name = hee_perm_freed
				duration = -1
				hidden = yes
			}
		}
	}
	
	option = {
	name = hee_prm.1.b
	}
}
