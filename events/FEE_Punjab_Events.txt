# Path to Jivanmukta
country_event = {
	id = FEE_Punjab_Events.1
	title = FEE_Punjab_Events.1.T
	desc = FEE_Punjab_Events.1.D
	picture = GOOD_WITH_MONARCH_eventPicture

	fire_only_once = yes

	trigger = {
		is_or_was_tag = { tag = PUN }
		NOT = {
			lahore_area = {
				type = all
				owned_by = ROOT
			}
		}
		NOT = {
			sirhind_area = {
				type = all
				owned_by = ROOT
			}
		}
		NOT = {
			sind_sagar_area = {
				type = all
				owned_by = ROOT
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = FEE_Punjab_Events.1.OPT1
		ai_chance = {
			factor = 75
		}

		add_mil_power = 25
		lahore_area = { add_permanent_claim = ROOT }
		sirhind_area = { add_permanent_claim = ROOT }
		sind_sagar_area = { add_permanent_claim = ROOT }
	}

	option = {
		name = FEE_Punjab_Events.1.OPT2
		ai_chance = {
			factor = 75
		}

		add_stability_or_adm_power = yes
		every_neighbor_country = {
			add_opinion = {
				who = ROOT
				modifier = opinion_eased_tension
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = opinion_eased_tension
			}
		}
	}
}

# The Sikh Confederacy
country_event = {
	id = FEE_Punjab_Events.2
	title = FEE_Punjab_Events.2.T
	desc = FEE_Punjab_Events.2.D
	picture = GOOD_WITH_MONARCH_eventPicture

	fire_only_once = yes

	trigger = {
		is_or_was_tag = { tag = PUN }
		religion = sikhism
		MUG = {
			is_neighbor_of = ROOT
			any_owned_province = {
				OR = {
					religion = sikhism
					culture = panjabi
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 24
	}

	option = {
		name = FEE_Punjab_Events.2.OPT1
		ai_chance = {
			factor = 75
		}

		add_yearly_manpower = 3
		add_army_tradition = 15
		add_country_modifier = {
			name = fee_pun_martyrdom
			duration = -1
			desc = FEE_PUN_MARTYRDOM_TT
		}
		custom_tooltip = FEE_PUN_MARTYRDOM_CLAIM_TT
		hidden_effect = {
			MUG = {
				every_owned_province = {
					limit = {
						OR = {
							religion = sikhism
							culture = panjabi
						}
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
}
