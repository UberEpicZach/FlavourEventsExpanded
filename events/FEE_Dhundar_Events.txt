namespace = FEE_Dhundar_Events

# The Founding of Jaipur and its school of Natural Philosophy
country_event = {
	id = FEE_Dhundar_Events.1
	title = FEE_Dhundar_Events.1.t
	desc = FEE_Dhundar_Events.1.d
	picture = CITY_VIEW_eventPicture

	fire_only_once = yes

	goto = 519

	trigger = {
		owns = 519
		ADM = 4
		natural_scientist = 3
		OR = {
			has_idea_group = innovativeness_ideas
			AND = {
				is_expanded_mod_active = { mod = historical_ideas }
				tag = DHU
				has_idea_group = HIE_DHU_ideas
				HIE_DHU_ideas = 3
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = FEE_Dhundar_Events.1.a

		519 = {
			add_base_tax = 2
			add_province_modifier = {
				name = fee_dhundar_natural_philosophy
				duration = -1
				desc = until_end_of_age_desc
			}
		}
		if = {
			limit = {
				ADM = 6
			}
			519 = {
				if = {
					limit = {
						NOT = { has_building = university }
					}
					add_building = university
				}
			}
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				add_splendor = 200
			}
			else = {
				add_prestige_or_monarch_power = { amount = 25 }
			}
		}
	}
}
